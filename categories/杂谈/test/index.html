#include<bits/stdc++.h>
using namespace std;
#define LL long long
const LL MAXN=1e7+5;
LL P,n,m;
LL fac[MAXN];
LL ksm(LL x,LL y){
    // cerr<<x<<" "<<y<<endl;
    LL res=1;
    while(y){
        if(y&1){
            res=res*x%P;
        }
        x=x*x%P;
        y>>=1;
    }
    return res;
}
int main(){
    scanf("%lld%lld%lld",&n,&m,&P);
    if(n==m){
        printf("0");
        return 0;
    }
    fac[0]=1;
    for(LL i=1;i<=1e7;i++){
        fac[i]=fac[i-1]*i%P;
    }
    LL ans=fac[n-1]*ksm(fac[m],P-2)%P*ksm(fac[n-1-m],P-2)%P;
    ans=ans*ksm(m,n-m-1)%P;
    ans=ans*ksm(n-m,m-1)%P;
    ans=ksm(n,n-2)*ksm(ans,P-2)%P;
    printf("%lld",ans);
}
